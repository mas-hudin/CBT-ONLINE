Integrasi Google Sheets (Database)

Gunakan Google Sheets sebagai server pusat untuk menyimpan semua data ujian secara permanen.

Cara Menghubungkan:

  1. Buka Google Sheets baru dan beri nama.
  2. Klik menu Ekstensi > Apps Script.
  3. Hapus kode yang ada, dan Paste Kode Script di bawah ini. Klik Save (ikon disket).
  4. Klik tombol biru Terapkan (Deploy) > Deployment Baru.
  5. Pilih jenis Aplikasi Web (Web App). Akses: pilih Siapa Saja (Anyone).
  6. Copy URL Web App yang dihasilkan, lalu Paste ke kolom di bawah.
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  // Setup header jika kosong
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["Tanggal Buat", "ID Ujian", "Mata Pelajaran", "Token", "Link Form", "Durasi (Menit)", "Kunci Browser", "Kamera"]);
    sheet.getRange("A1:H1").setFontWeight("bold").setBackground("#dbeafe");
  }

  try {
    var data = JSON.parse(e.postData.contents);
    if(data.action === "simpanUjian") {
      var d = data.payload;
      sheet.appendRow([new Date(), d.id, d.title, d.token, d.url, d.durationMinutes, d.security.lockdown, d.security.webcam]);
      return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "Data saved"}))
                           .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()}))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}

Jika dikosongkan, aplikasi akan menggunakan Local Storage browser (Offline Mode).
